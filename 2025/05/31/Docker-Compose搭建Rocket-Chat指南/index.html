Docker Compose搭建Rocket.Chat指南

version: ‘3.8’

services:
rocketchat:
image: rocket.chat:latest
restart: always
ports:
- “8080:3000” # 将宿主机的8080端口映射到容器3000端口
environment:
- PORT=3000
- ROOT_URL=http://IP:8080 # 使用您的公网IP和端口
- MONGO_URL=mongodb://mongo:27017/rocketchat?replicaSet=rs0
- MONGO_OPLOG_URL=mongodb://mongo:27017/local?replicaSet=rs0
depends_on:
- mongo
volumes:
- ./uploads:/app/uploads

mongo:
image: mongo:5.0
restart: always
command: mongod –replSet rs0 –oplogSize 128
volumes:
- ./data/db:/data/db
- ./data/dump:/dump
environment:
- TZ=Asia/Shanghai

mongo-init:
image: mongo:5.0
command: >
bash -c
“for i in seq 1 30; do
mongo mongo:27017 –eval "rs.initiate({_id:’rs0’, members:[{_id:0, host:’mongo:27017’}]})" > /dev/null 2>&1 &&
s=$$? && break || s=$$?;
echo "Tried $$i times. Waiting 5 sec…";
sleep 5;
done; (exit $$s)”
depends_on:
- mongo
